#!/usr/bin/env bash #Translation export TEXTDOMAINDIR="/usr/share/locale" export TEXTDOMAIN=biglinux-webapps echo '
' DESKTOP="$(ls $HOME/.local/share/applications/*webapp-biglinux-custom.desktop)" if [ "$DESKTOP" ];then echo '

'$"Remover WebApps BigLinux"'

' c=0 for i in $DESKTOP; do if [ "$(grep Exec= $i | grep firefox)" ];then DESKEXEC="$(grep Exec= $i | sed 's|Exec=||')" DESKURL="$(grep "\-new-instance" $DESKEXEC | cut -d' ' -f9 | sed 's|^"||;s|"$||')" else DESKURL="$(grep Exec= $i | sed 's|.*app=||')" fi DESKNAME="$(grep "Name=" $i | sed 's|Name=||')" DESKICON="$(grep "Icon=" $i | sed 's|Icon=||')" if [ "$DESKICON" = "webapp" ];then DESKICON="img/default.png" fi echo '
'$"Atenção!"'
'$"Você tem certeza que deseja remover o WebApp"'
"'$DESKNAME'" ?
' c=$((c+1)) done echo '
'$"ÍCONE"' '$"NOME"' '$"URL"' '$"AÇÃO"'
'$DESKNAME' '$DESKURL'
'$"Sucesso!"'
'$"O WebApp foi removido com sucesso!"'
'$"Erro!"'
'$"Não foi possível remover o WebApp!"'
'$"Por favor, verifique e tente novamente."'
' else echo '

'$"Remover WebApps BigLinux"'

'$"Atenção! Nenhum WebApp BigLinux instalado por você!"'

'$"Se você deseja instalar um novo WebApp BigLinux, clique no botão Instalar"'


'$"Instalar"'
' fi echo ''
